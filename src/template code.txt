// WRITING TEXT (IN RENDER FUNCTION)

SDL_Color textColor = {255, x, 255, 255};
SDL_Surface* textSurface = TTF_RenderText_Solid(font, "Hello, World!", (int)strlen("Hello, World!"), textColor);
SDL_Texture* textTexture = SDL_CreateTextureFromSurface(renderer, textSurface);
SDL_FRect renderQuad = {
    .x = 10.0f,
    .y = 10.0f,
    .w = (float)textSurface->w,
    .h = (float)textSurface->h
};
SDL_RenderTexture(renderer, textTexture, NULL, &renderQuad);

// DETECT IF KEYS ARE PRESSED

if (currKeyState[SDL_SCANCODE_RETURN] && !prevKeyState[SDL_SCANCODE_RETURN]) {
    printf("Enter key was just pressed\n");
}

// DETECT IF MOUSE IF CLICKED

if ((currMouseState & SDL_BUTTON_LMASK) && !(prevMouseState & SDL_BUTTON_LMASK)) {
